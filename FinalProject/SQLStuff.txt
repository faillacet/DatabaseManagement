CREATE DATABASE nbaproject;
USE nbaproject;

CREATE TABLE team (
teamID int primary key not null,
name varchar(30),
abbreviation varchar(4),
nickname varchar(20),
city varchar(25),
state varchar(25),
yearfounded int);

select * from team;

CREATE TABLE player (
playerID int primary key not null,
teamID int,
fullName varchar(40),
firstName varchar(20),
lastName varchar(20),
isActive bool,
FOREIGN KEY (teamID) REFERENCES team(teamID)
);

DROP TABLE player;
SELECT * FROM player;

CREATE TABLE playerstats (
playerID integer primary key not null,
timeframe varchar(15),
points float,
assists float,
rebounds float,
playerimpactestimate float,
FOREIGN KEY (playerID) REFERENCES player(playerID)
);

drop table playerstats;
select * from playerstats;

CREATE TABLE teamstats (
teamID int primary key not null,
timeframe varchar(15),
wins int,
losses int,
wPCT float,
fgPCT float,
fg3PCT float,
rebounds float,
assists float,
blocks float,
points float,
FOREIGN KEY (teamID) REFERENCES team(teamID)
);

DROP TABLE teamstats;
select * from teamstats;
